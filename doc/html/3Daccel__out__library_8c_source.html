318 void pwmAngleCalc(int16_t positionX, int16_t positionY, int16_t positionZ)
319 {
320  double roll, pitch;
321  double gX = 0, gY = 0, gZ = 0;
322  static double rollStored = 0.0;
323  static double pitchStored = 0.0;
324 
325  // calculate acceleration values from raw values
326  gX = positionX / GDIVIDER;
327  gY = positionY / GDIVIDER;
328  gZ = positionZ / GDIVIDER;
329 
330  // calculate angles for servos
331  roll = atan(gY/(sqrt((gX*gX)+(gZ*gZ)))) * 180 / PI;
332  pitch = atan(gX/(sqrt((gY*gY)+(gZ*gZ)))) * 180 / PI;
333 
334  // angle correction as documented
335  if (pitch >= 0.0)
336  {
337  pitch += pitch * PITCHCORRECTPOS;
338  }
339 
340  // angle thresholding
341  if (startup == 0)
342  {
343  pitchStored = pitch;
344  rollStored = roll;
345  }
346  else
347  {
348  if (averageChoice == 1)
349  {
350  if (fabs(pitch - pitchStored) < ANGAVERAGE)
351  {
352  pitch = pitchStored;
353  }
354  else
355  {
356  pitchStored = pitch;
357  }
358 
359  if (fabs(roll - rollStored) < ANGAVERAGE)
360  {
361  roll = rollStored;
362  }
363  else
364  {
365  pitchStored = pitch;
366  }
367  }
368  }
369 
370  startup++;
371 
372  // final signal values
373 
374  signal1 = (90.00+roll) * SERVOUOLINEAR + SERUPLO0;
375  signal2 = (90.00+pitch) * SERVOLOLINEAR + SERVOLO0;
376 
377  signal1 = (int)(signal1 * RNDFACTOR + 0.5) / RNDFACTOR;
378  signal2 = (int)(signal2 * RNDFACTOR + 0.5) / RNDFACTOR;
