272 void pwmAngleCalc(int16_t positionX, int16_t positionY, int16_t positionZ)
273 {
274  double pi = PI;
275  double divider = GDIVIDER;
276  double roll, pitch;
277  double gX, gY, gZ;
278 
279  gX = positionX / divider;
280  gY = positionY / divider;
281  gZ = positionZ / divider;
282 
283  roll = atan(gY/(sqrt((gX*gX)+(gZ*gZ)))) * 180 / pi;
284  pitch = atan(gX/(sqrt((gY*gY)+(gZ*gZ)))) * 180 / pi;
285 
286  if (pitch >= 0.0)
287  {
288  pitch += pitch * PITCHCORRECTPOS;
289  }
290 
291  signal1 = (90.00+roll) * SERVOUOLINEAR + SERUPLO0;
292  signal2 = (90.00+pitch) * SERVOLOLINEAR + SERVOLO0;
293 
294  signal1 = (int)(signal1 * 1000 + 0.5) / 1000.0;
295  signal2 = (int)(signal2 * 1000 + 0.5) / 1000.0;
296 }
