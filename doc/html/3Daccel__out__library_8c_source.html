270 void pwmAngleCalc(int16_t positionX, int16_t positionY, int16_t positionZ)
271 {
272  double pi = PI;
273  double divider = GDIVIDER;
274  double roll, pitch;
275  double gX, gY, gZ;
276 
277  gX = positionX / divider;
278  gY = positionY / divider;
279  gZ = positionZ / divider;
280 
281  roll = atan(gY/(sqrt((gX*gX)+(gZ*gZ)))) * 180 / pi;
282  pitch = atan(gX/(sqrt((gY*gY)+(gZ*gZ)))) * 180 / pi;
283 
284  if (pitch >= 0.0)
285  {
286  pitch += pitch * PITCHCORRECTPOS;
287  }
288 
289  signal1 = (90.00+roll) * SERVOUOLINEAR + SERUPLO0;
290  signal2 = (90.00+pitch) * SERVOLOLINEAR + SERVOLO0;
291 
292  signal1 = (int)(signal1 * RNDFACTOR + 0.5) / RNDFACTOR;
293  signal2 = (int)(signal2 * RNDFACTOR + 0.5) / RNDFACTOR;
294 }
