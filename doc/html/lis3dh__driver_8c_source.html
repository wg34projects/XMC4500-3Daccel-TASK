355 /*******************************************************************************
356 * Function Name : LIS3DH_SetMode
357 * Description : Sets LIS3DH Operating Mode
358 * Input : Modality (LIS3DH_NORMAL, LIS3DH_LOW_POWER, LIS3DH_POWER_DOWN)
359 * Output : None
360 * Return : Status [MEMS_ERROR, MEMS_SUCCESS]
361 *******************************************************************************/
362 status_t LIS3DH_SetMode(LIS3DH_Mode_t md) {
363  u8_t value;
364  u8_t value2;
365  static u8_t ODR_old_value;
366 
367  if( !LIS3DH_ReadReg(LIS3DH_CTRL_REG1, &value) )
368  return MEMS_ERROR;
369 
370  if( !LIS3DH_ReadReg(LIS3DH_CTRL_REG4, &value2) )
371  return MEMS_ERROR;
372 
373  if((value & 0xF0)==0)
374  value = value | (ODR_old_value & 0xF0); //if it comes from POWERDOWN
375 
376  switch(md) {
377 
378  case LIS3DH_POWER_DOWN:
379  ODR_old_value = value;
380  value &= 0x0F;
381  break;
382 
383  case LIS3DH_NORMAL:
384  value &= 0xF7;
385  value |= (MEMS_RESET<<LIS3DH_LPEN);
386  value2 &= 0xF7;
387  value2 |= (MEMS_SET<<LIS3DH_HR); //set HighResolution_BIT
388  break;
389 
390  case LIS3DH_LOW_POWER:
391  value &= 0xF7;
392  value |= (MEMS_SET<<LIS3DH_LPEN);
393  value2 &= 0xF7;
394  value2 |= (MEMS_RESET<<LIS3DH_HR); //reset HighResolution_BIT
395  break;
396 
397  default:
398  return MEMS_ERROR;
399  }
400 
401  if( !LIS3DH_WriteReg(LIS3DH_CTRL_REG1, value) )
402  return MEMS_ERROR;
403 
404  if( !LIS3DH_WriteReg(LIS3DH_CTRL_REG4, value2) )
405  return MEMS_ERROR;
406 
407  return MEMS_SUCCESS;
408 }
