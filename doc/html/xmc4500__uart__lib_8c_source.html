48 void _init_uart0_ch0()
49 {
50  /* USIC channels initialization */
51  XMC_UART_CH_Init (XMC_UART0_CH0, &uart_config);
52 
53  XMC_UART_CH_SetInputSource (XMC_UART0_CH0, XMC_UART_CH_INPUT_RXD,
54  USIC0_C0_DX0_P1_4);
55 
56  /* FIFOs initialization for both channels:
57  * 8 entries for TxFIFO from point 0, LIMIT=1
58  * 8 entries for RxFIFO from point 8, LIMIT=7 (SRBI is set if all 8*data have
59  * been received)
60  * */
61  XMC_USIC_CH_TXFIFO_Configure (XMC_UART0_CH0, 0, XMC_USIC_CH_FIFO_SIZE_8WORDS, TX_FIFO_INITIAL_LIMIT);
62  XMC_USIC_CH_RXFIFO_Configure (XMC_UART0_CH0, 8, XMC_USIC_CH_FIFO_SIZE_8WORDS, RX_FIFO_INITIAL_LIMIT);
63 
64  /* Enabling events for TX FIFO and RX FIFO */
65  XMC_USIC_CH_RXFIFO_EnableEvent (XMC_UART0_CH0,
66  XMC_USIC_CH_RXFIFO_EVENT_CONF_STANDARD |
67  XMC_USIC_CH_RXFIFO_EVENT_CONF_ALTERNATE);
68 
69  /* Connecting the previously enabled events to a Service Request line number */
70  XMC_USIC_CH_RXFIFO_SetInterruptNodePointer (XMC_UART0_CH0, XMC_USIC_CH_RXFIFO_INTERRUPT_NODE_POINTER_STANDARD, 0);
71  XMC_USIC_CH_RXFIFO_SetInterruptNodePointer (XMC_UART0_CH0, XMC_USIC_CH_RXFIFO_INTERRUPT_NODE_POINTER_ALTERNATE, 0);
72 
73  /* Start USIC operation as UART */
74  XMC_UART_CH_Start (XMC_UART0_CH0);
75 
76  /*Initialization of the necessary ports*/
77  XMC_GPIO_Init (UART_TX, &uart_tx);
78  XMC_GPIO_Init (UART_RX, &uart_rx);
79 
80  /* Configuring priority and enabling NVIC IRQ for the defined service request
81  line number */
82  NVIC_SetPriority (USIC0_0_IRQn, 63U);
83  NVIC_EnableIRQ (USIC0_0_IRQn);
84 
85  return;
86 }

97  if ( (rx_tmp != STRING_CR) && (!str_available)) {
